	}
	title = bar_sane_ambrosius.7.t
	desc = bar_sane_ambrosius.7.d
	picture = TO_CHANGE
	is_triggered_only = yes
	option = {
		name = bar_sane_ambrosius.7.a
		add_ideas = modern_society
		hidden_effect = {
			kill_country_leader = yes
			sound_effect = bar_amb_gunshot
			set_variable = {
				bar_amb_gunshot_visible = 1
			}
		}
		add_country_leader_role = {
			character = BAR_gasienica
			promote_leader = yes
			country_leader = {
				expire = "1965.1.1"
				ideology = despotism
			}
		}
		if = { #democratic ending 
			limit = {
				has_completed_focus =  BAR_amb_commonwealth
			}
			if = { #technocrato-oligarchic ending
				limit = {
					power_balance_value = {
						id = bar_sane_ambrosius_balance
						value > 0.89
					}
				}
				country_event = { id = bar_sane_ambrosius.1009 days = 7 }
			} else_if = { #cultists democracy
				limit = {
					power_balance_value = {
						id = bar_sane_ambrosius_balance
						value < 0.89
					}
				}
				country_event = { id = bar_sane_ambrosius.1010 days = 7 }
			} else = { #regular party system
				country_event = { id = bar_sane_ambrosius.1011 days = 7 }
			}
		} else = { #kingdom ending
			if = { #gąsienica becomes regent for life as nobody makes claim for crown
				limit = {
					power_balance_value = {
						id = bar_sane_ambrosius_balance
						value > 0.89
					}
				} 
				country_event = { id = bar_sane_ambrosius.1012 days = 7 }
			} else_if = { #cultist monarch
				limit = {
					power_balance_value = {
						id = bar_sane_ambrosius_balance
						value < 0.89
					}
				}
				country_event = { id = bar_sane_ambrosius.1013 days = 7 }
			} else = { #duels for crown continue
				country_event = { id = bar_sane_ambrosius.1014 days = 7 }
			}
		}
	}
}
